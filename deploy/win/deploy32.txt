Обязательно удалить каталог сборки qt
cd "C:\Qt\5.10.1\msvc2015\bin"
.\windeployqt.exe D:\qt\MetaGate\MetaGate.exe

quazip-0.7.3/libs/win32/quazip.dll
secp256k1/lib/windows32/secp256k1.dll
и папку
